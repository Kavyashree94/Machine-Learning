def compute_cost(x, y, w, b):
    m = x.shape[0]
    cost_sum = 0.0

    for i, x_i in enumerate(x):
        f_wb = w * x_i + b
        cost = (f_wb - y[i]) ** 2
        cost_sum += cost

    return cost_sum / (2 * m)
